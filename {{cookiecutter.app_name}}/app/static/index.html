# create a basket
curl -X POST \
    -u AzureDiamond:hunter2 \
    -H 'Content-Type: application/json' \
    http://127.0.0.1:5000/basket \
    -d '{"name":"Hello Basket"}'

# create a foo, and relate a basket to it
curl -X POST \
    -u AzureDiamond:hunter2 \
    -H 'Content-Type: application/json' \
    http://127.0.0.1:5000/foo \
    -d '{"bar":42,"baz":"quux","basket_id":1}'

# read it back
curl -X GET \
    -u AzureDiamond:hunter2 \
    -H 'Accept: application/json' \
    http://127.0.0.1:5000/foo/1
